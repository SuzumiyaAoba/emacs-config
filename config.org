* Startup

ファイル指定なしで起動したときは、起動ディレクトリを `dired` で開く。

#+begin_src emacs-lisp :tangle init.el
(defun my/open-startup-directory-in-dired ()
  "Open the startup directory in Dired when Emacs starts on *scratch*."
  (when (and (string= (buffer-name) "*scratch*")
             (file-directory-p default-directory))
    (switch-to-buffer (dired-noselect default-directory))))

(add-hook 'window-setup-hook #'my/open-startup-directory-in-dired)
#+end_src

* Tree-sitter

tree-sitter によるシンタックスハイライトを優先して使う。

#+begin_src emacs-lisp :tangle init.el
(defun my/treesit-language-ready-p (language)
  "Return non-nil when LANGUAGE can be used by tree-sitter."
  (and (fboundp 'treesit-available-p)
       (treesit-available-p)
       (fboundp 'treesit-language-available-p)
       (treesit-language-available-p language)))

(setq treesit-font-lock-level 4)

(dolist (entry '((bash sh-mode bash-ts-mode)
                 (c c-mode c-ts-mode)
                 (cpp c++-mode c++-ts-mode)
                 (css css-mode css-ts-mode)
                 (html mhtml-mode html-ts-mode)
                 (java java-mode java-ts-mode)
                 (javascript javascript-mode js-ts-mode)
                 (json js-json-mode json-ts-mode)
                 (python python-mode python-ts-mode)
                 (toml conf-toml-mode toml-ts-mode)))
  (pcase-let ((`(,language ,mode ,ts-mode) entry))
    (when (and (fboundp ts-mode)
               (my/treesit-language-ready-p language))
      (add-to-list 'major-mode-remap-alist (cons mode ts-mode)))))

(dolist (entry '((cmake "CMakeLists\\.txt\\'" cmake-ts-mode)
                 (cmake "\\.cmake\\'" cmake-ts-mode)
                 (dockerfile "Dockerfile\\'" dockerfile-ts-mode)
                 (dockerfile "\\.dockerfile\\'" dockerfile-ts-mode)
                 (go "\\.go\\'" go-ts-mode)
                 (rust "\\.rs\\'" rust-ts-mode)
                 (tsx "\\.tsx\\'" tsx-ts-mode)
                 (typescript "\\.ts\\'" typescript-ts-mode)
                 (yaml "\\.ya?ml\\'" yaml-ts-mode)))
  (pcase-let ((`(,language ,pattern ,mode) entry))
    (when (and (fboundp mode)
               (my/treesit-language-ready-p language))
      (add-to-list 'auto-mode-alist (cons pattern mode)))))
#+end_src

* Icons

GUI の `dired` では `nerd-icons-dired` を有効にする。

#+begin_src emacs-lisp :tangle init.el
(defun my/enable-nerd-icons-dired-mode ()
  "Enable `nerd-icons-dired-mode' in graphical Dired buffers."
  (when (display-graphic-p)
    (nerd-icons-dired-mode 1)))

(packlet nerd-icons-dired
  :autoload nerd-icons-dired-mode
  :hook (dired-mode-hook . my/enable-nerd-icons-dired-mode))
#+end_src

* Undo

`undohist` で undo 履歴を保存し、`vundo` で可視化する。

#+begin_src emacs-lisp :tangle init.el
(packlet undohist
  :init
  (setq undohist-directory
        (expand-file-name "undohist/" user-emacs-directory))
  :autoload undohist-initialize
  :hook (after-init-hook . undohist-initialize))

(packlet vundo
  :commands (vundo)
  :bind ("C-x u" . vundo))
#+end_src

* Git

`magit` を遅延ロードで使えるようにする。

#+begin_src emacs-lisp :tangle init.el
(packlet magit
  :init
  (setq magit-define-global-key-bindings nil
        magit-status-show-untracked-files nil
        magit-diff-refine-hunk nil
        magit-log-section-commit-count 10
        magit-refresh-status-buffer nil
        magit-save-repository-buffers 'dontask
        magit-status-sections-hook
        '(magit-insert-status-headers
          magit-insert-unstaged-changes
          magit-insert-staged-changes))
  :commands (magit-status magit-dispatch)
  :bind (("C-x g" . magit-status)
         ("C-x M-g" . magit-dispatch)))
#+end_src
