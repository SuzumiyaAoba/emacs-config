#+begin_src emacs-lisp :tangle init.el
;; init.el --- Minimal emacs-batteries setup -*- lexical-binding: t; coding: utf-8 -*-

(defconst my/config-root
  (file-name-as-directory
   (expand-file-name
    (or (getenv "EMACS_CONFIG_ROOT")
        (file-name-directory (or load-file-name user-init-file))))))

(let ((root (expand-file-name user-emacs-directory))
      kept)
  (dolist (dir load-path)
    (unless (and (stringp dir)
                 (string-equal (file-name-as-directory (expand-file-name dir))
                               root))
      (push dir kept)))
  (setq load-path (nreverse kept)))

(add-to-list 'load-path
             (expand-file-name "site-lisp/packlet" my/config-root))
(add-to-list 'load-path
             (expand-file-name "site-lisp/emacs-batteries" my/config-root))

(require 'packlet)
(require 'emacs-batteries)
(emacs-batteries-setup)

* Input method

=DDSKK= を日本語入力として使う。

#+begin_src emacs-lisp :tangle init.el
(packlet skk
  :init
  (setq default-input-method "japanese-skk"
        skk-user-directory (expand-file-name "skk/" user-emacs-directory))
  (make-directory skk-user-directory t)
  (autoload 'ccc-setup "ccc" nil t)
  (autoload 'skk-cus-setup "skk-cus" nil t)
  (autoload 'skk-cursor-current-color "skk-cursor" nil t)
  (autoload 'skk-cursor-set-1 "skk-cursor" nil t)
  (autoload 'skk-cursor-off-1 "skk-cursor" nil t)
  (autoload 'skk-emacs-prepare-menu "skk-emacs" nil t)
  (autoload 'skk-annotation-start-python "skk-annotation" nil t)
  (autoload 'skk-version "skk-version" nil t)
  (autoload 'skk-activate "skk-leim" nil t)
  (autoload 'skk-auto-fill-activate "skk-leim" nil t)
  (defun my/load-ddskk-autoloads ()
    "Load `ddskk-autoloads' while skipping the obsolete isearch advice form."
    (when-let ((file (locate-library "ddskk-autoloads")))
      (with-temp-buffer
        (insert-file-contents file)
        (goto-char (point-min))
        (let (form)
          (condition-case nil
              (while t
                (setq form (read (current-buffer)))
                (unless (and (listp form)
                             (eq (car-safe form) 'defconst)
                             (eq (cadr form) 'skk-isearch-really-early-advice))
                  (eval form t)))
            (end-of-file nil))))))
  (my/load-ddskk-autoloads)
  (register-input-method
   "japanese-skk" "Japanese" #'skk-activate ""
   "Simple Kana to Kanji conversion program")
  (register-input-method
   "japanese-skk-auto-fill" "Japanese" #'skk-auto-fill-activate ""
   "Simple Kana to Kanji conversion program with auto-fill")
  (defun my/skk-handle-c-j ()
    "Enable SKK or return to hiragana mode on `C-j'."
    (interactive)
    (cond
     ((not (bound-and-true-p skk-mode))
      (let ((input-method-verbose-flag nil))
        (activate-input-method default-input-method)))
     ((bound-and-true-p skk-henkan-mode)
      (call-interactively #'skk-kakutei))
     ((or (bound-and-true-p skk-latin-mode)
          (bound-and-true-p skk-jisx0208-latin-mode)
          (bound-and-true-p skk-abbrev-mode)
          (bound-and-true-p skk-katakana))
      (skk-j-mode-on))
     (t
      (call-interactively #'skk-kakutei))))
  (defvar my/skk-shortcut-mode-map
    (let ((map (make-sparse-keymap)))
      (define-key map (kbd "C-j") #'my/skk-handle-c-j)
      (define-key map (kbd "C-x j") #'toggle-input-method)
      map))
  (define-minor-mode my/skk-shortcut-mode
    "Provide global keybindings for SKK."
    :global t
    :lighter ""
    :keymap my/skk-shortcut-mode-map)
  (defun my/skk-auto-enable-p ()
    "Return non-nil when SKK should be enabled automatically."
    (and (not noninteractive)
         (not (minibufferp))
         (not buffer-read-only)
         (not (derived-mode-p 'dired-mode))
         (not (derived-mode-p 'special-mode))))
  (defvar my/skk-default-enable-delay 1.0
    "Idle seconds before enabling SKK by default.")
  (defun my/enable-skk-by-default ()
    "Enable SKK by default in editable interactive buffers.
When SKK is newly enabled, switch to ASCII mode."
    (cond
     ((my/skk-auto-enable-p)
      (when (null current-input-method)
        (let ((input-method-verbose-flag nil))
          (activate-input-method default-input-method))
        (when (and (bound-and-true-p skk-mode)
                   (fboundp 'skk-latin-mode))
          (skk-latin-mode 1))))
     ((and (stringp current-input-method)
           (string-prefix-p "japanese-skk" current-input-method))
      (let ((input-method-verbose-flag nil))
        (deactivate-input-method)))))
  (defun my/schedule-skk-by-default ()
    "Schedule delayed SKK activation for the current buffer."
    (let ((buffer (current-buffer)))
      (run-with-idle-timer
       my/skk-default-enable-delay nil
       (lambda (buffer)
         (when (buffer-live-p buffer)
           (with-current-buffer buffer
             (my/enable-skk-by-default))))
       buffer)))
  (my/skk-shortcut-mode 1)
  :hook ((after-init-hook . my/schedule-skk-by-default)
         (after-change-major-mode-hook . my/schedule-skk-by-default))
  :config
  (with-eval-after-load 'skk
    (when window-system
      (ccc-setup)))
  :commands (skk-mode))
#+end_src

(defun my/consult-fd-current-directory-items ()
  "Return non-hidden entries in `default-directory' for `my/consult-fd'."
  (sort
   (mapcar
    (lambda (name)
      (if (file-directory-p (expand-file-name name default-directory))
          (concat name "/")
        name))
    (directory-files default-directory nil directory-files-no-dot-files-regexp))
   #'string-lessp))

(defun my/consult-fd-open (dir file)
  "Open FILE relative to DIR.
When FILE is empty, open DIR itself."
  (consult--file-action
   (expand-file-name (if (string-empty-p file) "." file) dir)))

(defun my/consult-read-directory (prompt dir)
  "Return directory for consult commands.
Use DIR when given, otherwise use `default-directory'.
With `C-u', use the project root.  With larger prefix arguments,
prompt with PROMPT starting from `default-directory'."
  (file-name-as-directory
   (expand-file-name
    (cond
     ((stringp dir) dir)
     ((equal dir '(4)) (or (consult--project-root) default-directory))
     (dir (read-directory-name prompt default-directory nil t))
     (t default-directory)))))

(defun my/consult-fd (&optional dir initial)
  "Select a file in DIR, showing current directory entries and `fd' results."
  (interactive "P")
  (require 'consult)
  (let* ((dir (my/consult-read-directory "Fd directory: " dir))
         (prompt (format "Fd (%s): " (consult--left-truncate-file dir)))
         (paths '(".")))
    (pcase-let* ((default-directory dir)
                 (builder (consult--fd-make-builder paths))
                 (sources
                  `((:name "Current"
                     :narrow ?c
                     :category file
                     :face consult-file
                     :history file-name-history
                     :state ,#'consult--file-preview
                     :action ,(lambda (file)
                                (my/consult-fd-open dir file))
                     :new ,(lambda (file)
                             (my/consult-fd-open dir file))
                     :items ,#'my/consult-fd-current-directory-items)
                    (:name "fd"
                     :narrow ?f
                     :category file
                     :face consult-file
                     :history file-name-history
                     :state ,#'consult--file-preview
                     :action ,(lambda (file)
                                (my/consult-fd-open dir file))
                     :new ,(lambda (file)
                             (my/consult-fd-open dir file))
                     :async
                     ,(consult--process-collection
                       builder
                       :transform (consult--async-map
                                   (lambda (x) (string-remove-prefix "./" x)))
                       :highlight t
                       :file-handler t)))))
      (consult--multi
       sources
       :prompt prompt
       :sort nil
       :require-match nil
       :initial initial
       :add-history (thing-at-point 'filename)
       :history '(:input consult--find-history)))))

(defun my/consult-ripgrep (&optional dir initial)
  "Search with `rg' in DIR, defaulting to the project root."
  (interactive "P")
  (require 'consult)
  (consult-ripgrep
   (file-name-as-directory
    (expand-file-name
     (cond
      ((stringp dir) dir)
      (dir (read-directory-name "Ripgrep directory: " default-directory nil t))
      (t (or (consult--project-root) default-directory)))))
   initial))

* Tree-sitter

tree-sitter によるシンタックスハイライトを優先して使う。

#+begin_src emacs-lisp :tangle init.el
(defun my/treesit-language-ready-p (language)
  "Return non-nil when LANGUAGE can be used by tree-sitter."
  (and (fboundp 'treesit-available-p)
       (treesit-available-p)
       (fboundp 'treesit-language-available-p)
       (treesit-language-available-p language)))

(setq treesit-font-lock-level 4)

(dolist (entry '((bash sh-mode bash-ts-mode)
                 (c c-mode c-ts-mode)
                 (cpp c++-mode c++-ts-mode)
                 (css css-mode css-ts-mode)
                 (html mhtml-mode html-ts-mode)
                 (java java-mode java-ts-mode)
                 (javascript javascript-mode js-ts-mode)
                 (json js-json-mode json-ts-mode)
                 (python python-mode python-ts-mode)
                 (toml conf-toml-mode toml-ts-mode)))
  (pcase-let ((`(,language ,mode ,ts-mode) entry))
    (when (and (fboundp ts-mode)
               (my/treesit-language-ready-p language))
      (add-to-list 'major-mode-remap-alist (cons mode ts-mode)))))

(dolist (entry '((cmake "CMakeLists\\.txt\\'" cmake-ts-mode)
                 (cmake "\\.cmake\\'" cmake-ts-mode)
                 (dockerfile "Dockerfile\\'" dockerfile-ts-mode)
                 (dockerfile "\\.dockerfile\\'" dockerfile-ts-mode)
                 (go "\\.go\\'" go-ts-mode)
                 (rust "\\.rs\\'" rust-ts-mode)
                 (tsx "\\.tsx\\'" tsx-ts-mode)
                 (typescript "\\.ts\\'" typescript-ts-mode)
                 (yaml "\\.ya?ml\\'" yaml-ts-mode)))
  (pcase-let ((`(,language ,pattern ,mode) entry))
    (when (and (fboundp mode)
               (my/treesit-language-ready-p language))
      (add-to-list 'auto-mode-alist (cons pattern mode)))))
#+end_src
